def remove_duplicates(lst):
    # Create a new list to store the unique elements
    unique_lst = []
    
    # Loop through the original list and add each element to the unique list
    for elem in lst:
        if elem not in unique_lst:
            unique_lst.append(elem)
    
    # Return the unique list
    return unique_lst

def removeDuplicates(lst):
    if not lst: return []
    n = len(lst)
    for i in range(n):
        if lst[i] != lst[i+1]: lst[i+1] = lst[i]
    return lst[:n]
